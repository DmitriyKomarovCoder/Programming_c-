
    while ((opt = getopt_long(argc, argv, "t:r:b:d:n::", long_options, nullptr)) != -1) {
        switch (opt) {
            case 't':
                fileNameTitle = optarg;
                break;
            case 'r':
                fileNameRatings = optarg;
                break;
            case 'b':
                fileNameBasics = optarg;
                break;
            case 'd':
                date = optarg;
                break;
            case 'n':
                if (optarg != nullptr) {
                    numberMoviesString = optarg;
                }
                break;
            default:
                std::cerr << "Invalid option" << std::endl;
                return 1;
        }
    }
